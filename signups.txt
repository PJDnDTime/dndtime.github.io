<p>Register to play in this Friday's episode of <u><b>Pete & Jeremy's DnDTime</b></u><b></b>. We reserve game one each week to prioritize new players, so don't be afraid if you've never played with us before! For the other games, we choose players based on their <b>points</b>. Every time a player participates in <u><b>Pete & Jeremy's DnDTime</b></u><b></b>, their point total is halved. <script>
function showhide() {
	var date = new Date();
	var ddate = date.getDay();
	var signupON = document.getElementById('SignupFormON');
	var signupOFF = document.getElementById('SignupFormOFF');
	signupON.style.display = (signupON.style.display == 'block') && ((ddate == 4) || (ddate == 5) || (ddate == 6)) ? 'none' : 'block';
	signupOFF.style.display = (signupOFF.style.display == 'none') && ((ddate == 4) || (ddate == 5) || (ddate == 6)) ? 'block' : 'none';
}
showhide();

		}
	});
}
